git config --global user.name - імя користувача

git config --global user.email - елактронна пошта користувача

git init - Ініціалізувати порожній локальний Git-репозиторій

git status - перевірити стан(також аоказує які файли добавленні до гіт)

git add - додає файли (git add . - додає всі файли)

git commit - зберегти зміни

git log - показує історію

git branch - створює вітка

git push - Покласти зміни з локального у віддалений репозиторій

git pull - Оновити дані (забрати зміни) з віддаленого репозиторію

git merge - обєднує вітки

git clone - клонувати репрозорій

git ignore - визначає навмисно невідстежувані файли, які потрібно ігнорувати

git checkout - переміщення між вітками
